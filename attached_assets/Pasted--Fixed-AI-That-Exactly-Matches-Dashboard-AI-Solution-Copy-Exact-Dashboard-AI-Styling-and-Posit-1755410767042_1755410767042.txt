  activeTab,
  chatMessages, 
  chatMessage, 
  setChatMessage,

  handleKeyPress

  chatMessages: any[];
  chatMessage: string;
  setChatMessage: (value: string) => void;

              <div className="mb-2">
                <h3 className="text-sm font-medium text-slate-700">Project Name</h3>
              </div>

          </div>

          {/* AI Assistant - EXACT Dashboard component */}
          <div className="flex justify-center">
